# MyBatis总结

# 第一步：MyBatis依赖包

要使用MyBatis持久层框架的话，首先必须在pom文件中导入MyBatis依赖包


<!-- Mybatis核心 -->
<dependency>
    <groupId>org.mybatis</groupId>
    <artifactId>mybatis</artifactId>
    <version>3.5.7</version>
</dependency>
<!--pagehelper分页插件依赖-->
<dependency>
    <groupId>com.github.pagehelper</groupId>
    <artifactId>pagehelper</artifactId>
    <version>5.2.0</version>
</dependency>

# 第二步：创建MyBatis的核心配置文件

MyBatis核心配置文件可以在官网上查看

jdbc.driver=com.mysql.cj.jdbc.Driver
jdbc.url=jdbc:mysql://localhost:3306/smbms02?serverTimezone=UTC
jdbc.username=root
jdbc.password=root

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
		PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
		"http://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
	<!--
    Mybatis核心配置文件中的标签必须要按照指定的顺序配置:
    properties?,settings?,typeAliases?,typeHandlers?,
    objectFactory?,objectWrapperFactory?,reflectorFactory?,
    plugins?,environments?,databaseIdProvider?,mappers?
-->
	<!--导入外部数据源配置-->
	<properties resource="jdbc.properties"/>

	<settings>
		<!--将下划线映射为驼峰-->
		<setting name="mapUnderscoreToCamelCase" value="true"/>
		<!--开启延迟加载 (懒加载)-->
		<setting name="lazyLoadingEnabled" value="true"/>
		<!--按需加载 (true:全部加载 false:按需求加载数据)-->
		<setting name="aggressiveLazyLoading" value="false"/>
        <!--全局自动映射-->
         <setting name="autoMappingBehavior" value="FULL"/>
	</settings>

	<typeAliases>
        <!--以包的方式来设置别名-->
		<package name="com.mybatis.pojo"/>
	</typeAliases>

	<plugins>
        <!--引入分页插件 (前提是必须在pom文件中导入分页的插件)-->
		<plugin interceptor="com.github.pagehelper.PageInterceptor"/>
	</plugins>
	<!--development：表示开发者模块  也可以添加test 测试模块-->
	<environments default="development">
		<environment id="development">
           	<!--JDBC：表示JDBC模式-->
			<transactionManager type="JDBC"/>
			<dataSource type="POOLED">
				<property name="driver" value="${jdbc.driver}"/>
				<property name="url" value="${jdbc.url}"/>
				<property name="username" value="${jdbc.username}"/>
				<property name="password" value="${jdbc.password}"/>
			</dataSource>
		</environment>
	</environments>
	<!--mapper映射 这里用的是以包的方式来映射mapper文件-->
	<mappers>
		<package name="com.mybatis.mapper"/>
	</mappers>
</configuration>

# 第四步：创建mapper接口和创建mapper.xml文件

创建与pojo包同级的mapper包在mapper包中定义接口
	关于多对一和一对多
		谨记：多对一在多里面创建一 ---> 创建的一是对象
			 一对多在一里面创建多 ---> 创建的多是List集合

public interface UserMapper{
    //查询所有方法
    public List<User> getUserAll();
    //增加方法
    public void addUser(User user);
    //修改方法
    public void updUser(User user);
    //删除方法
    public void delUser(Integer uId);
    //多对一查询 联表查询
    public List<User> getUserAndRole();
    //多对一查询 分步查询一
    public List<User> getUserAndRoleOne();
    //一对多查询 分步查询二
    public List<User> getRoleAndUserTwo(Integer rid);
}

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.mybatis.mapper.UserMapper">
    <!--查询所有方法-->
    <select id="getUserAll" resultTyype="User">
        select * from user
    </select>
    <!--增加方法-->
    <insert id="addUser">
    	insert into user values(#{id},#{name},#{rId})
    </insert>
    <!--严谨点的增加方法 trim：prefix表示在方法前面加个什么东西 suffix是在方法后面加什么,后面加上Overrides意思就是在什么之前或之后去掉什么-->
    <insert id="addUser">
    	insert into user 
        <trim prefix="(" suffixOverrides="," suffix=")">
            <if test="id != null">
            	id,
            </if>
            <if test="name != null">
            	name,
            </if>
            <if test="rId != null">
            	rid
            </if>
        </trim>
        values
        <trim prefix="(" suffixOverrides="," suffix=")">
        	<if test="id != null">
            	#{id},
            </if>
            <if test="name != null">
            	#{name},
            </if>
            <if test="rId != null">
            	#{rId}
            </if>
        </trim>
    </insert>
    <!--修改方法 <set>:可以去掉最后一个逗号  <where>:可以去掉多余的and-->
    <update id="updUser">
    	update user 
        <set>
        	name = #{name},
            rid = #{rId}
        </set>
        <where>
        	id = #{id}
        </where>
    </update>
    <!--删除方法-->
    <delete id="delUser">
        delete from user
        <where>
        	id = #{id}
        </where>
    </delete>
    <!--多对一查询 联表查询-->
    <sql id="sqlUser">
    	u.id,u.name,u.rid,r.id,r.name as rName
    </sql>
    <resultMap id="UserAndRoleMap" type="User">
        <!--property:表示是在User类中Role的属性名 javaType:表示在User类中Role属性名的类型-->
    	<assocaiation property="roles" javaType="role">
        	<result column="rName" property="name"></result>
        </assocaiation>
    </resultMap>
    <select id="getUserAndRole" resultMap="UserAndRoleMap">
    	select <include refid="sqlUser"/> from user as u
        inner join role as r
        on u.rid = r.id
    </select>
    <!--多对一查询 分步查询一-->
    <resultMap id="UserAndRoleMapOne" type="User">
    	<assocaiation property="roles" javaType="role" column="rid"
                      select="com.mybatis.mapper.UserMapper.getUserAndRoleTwo"></assocaiation>
    </resultMap>
    <select id="getUserAndRoleOne" resultMap="UserAndRoleMapOne">
    	select * from user
    </select>
    <!--一对多查询 分步查询二-->
    <select id="getRoleAndUserTwo" resultType="User">
    	select * from user where rid = #{rid}
    </select>
</mapper>

public interface RoleMapper{
    //多对一查询 分步查询二
    public List<Role> getUserAndRoleTwo(Integer rId);
    //一对多查询 联表查询
    public List<Role> getRoleAndUser();
    //一对多查询 分步查询一
    public List<Role> getRoleAndUserOne();
}

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.mybatis.mapper.RoleMapper">
    <!--多对一查询 分步查询二-->
    <select id="getUserAndRoleTwo" resultType="Role">
    	select * from role where id = #{rid}
    </select>
    <!--一对多查询 联表查询-->
    <sql id="sqlRole">
    	u.id,u.name as uName,u.rid,r.id,r.name
    </sql>
    <resultMap id="RoleAndUser" type="Role">
    	<collection property="users" ofType="User">
        	<result property="uName" column="name"></result>
        </collection>
    </resultMap>
    <select id="getRoleAndUser" resultMap="RoleAndUser">
    	select <include refid="sqlRole"/> from Role as r
        inner join User as u
        on r.id = u.rid
    </select>
    <!--一对多查询 -->
    <resultMap id="RoleAndUserOne" type="Role">
    	<collection property="users" ofType="User" column="id"
                    select="com.mybatis.mapper.RoleMapper.getRoleAndUserTwo"></collection>
    </resultMap>
    <select id="getRoleAndUserOne" resultMap="RoleAndUserOne">
        select * from role
    </select>
</mapper>

# 动态SQL

<!--if标签的使用 test中放的是if的条件。 注意事项：并(and),或(or) 这里&&和||已失效-->
<if test=""></if>
<!--where标签的使用 此标签可以去除多余的and，如果条件都未满足if 这里where将不会显现-->
<where>
	<if test="name != null and name != ''">
    	name = #{name}
    </if>
</where>
<!--
trim标签的使用，
prefix表示在trim标签前添加什么什么
prefixOverrides表示在trim标签后添加什么什么
suffix表示在trim标签前去除什么什么
suffixOverrides表示在trim标签后去除什么什么
-->
<trim prefix="where" prefixOverrides="and" suffix=";" suffixOverrides="and"></trim>
<!--choose、when、otherwise标签的使用  注意事项，这里只能有一个满足条件，只可运行一个标签-->
<choose>
	<when test="">
    </when>
    <when test="">
    </when>
    <!--这里代表switch的case,满足条件即运行when中的条件-->
    ...
    <otherwise>
        <!--这里代表switch的default-->
    </otherwise>
</choose>
<!--foreach标签的使用
collection里写对象名
item里写别名
separator表示以什么来进行分隔
open表示以什么开始
close表示以什么结尾
注意事项：open和close是指整个foreach的开始和结束-->
<foreach collection="user" item="u" separator="," open="(" close=")"></foreach>
<!--SQL片段标签的使用
主要功能：可以把sql要展示的字段来进行封装 谁用谁调取sql片段的id名-->
<sql id="userSql">
	u.id,u.name as uName
</sql>
<select id="getAll" resultType="user">
	select <include refid="userSql"></include>
    from user
</select>

# 分页插件

a>在查询功能之前使用PageHelper.startPage(int pageNum, int pageSize)开启分页功能

pageNum：当前页的页码

pageSize：每页显示的条数


b>在查询获取list集合之后，使用PageInfo pageInfo = new PageInfo<>(List list, int navigatePages)获取分页相关数据

list：分页之后的数据

navigatePages：导航分页的页码数


c>分页相关数据

PageInfo{

pageNum=8, pageSize=4, size=2, startRow=29, endRow=30, total=30, pages=8,

list=Page{count=true, pageNum=8, pageSize=4, startRow=28, endRow=32, total=30,

pages=8, reasonable=false, pageSizeZero=false},

prePage=7, nextPage=0, isFirstPage=false, isLastPage=true, hasPreviousPage=true,

hasNextPage=false, navigatePages=5, navigateFirstPage4, navigateLastPage8,

navigatepageNums=[4, 5, 6, 7, 8]

}

pageNum：当前页的页码

pageSize：每页显示的条数

size：当前页显示的真实条数

total：总记录数

pages：总页数

prePage：上一页的页码

nextPage：下一页的页码

isFirstPage/isLastPage：是否为第一页/最后一页

hasPreviousPage/hasNextPage：是否存在上一页/下一页

navigatePages：导航分页的页码数

navigatepageNums：导航分页的页码，[1,2,3,4,5]
